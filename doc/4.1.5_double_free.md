## 4.1.5 double_free
write basic framework
```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

from pwn import *
from time import sleep
context.log_level = "debug"
context.binary = "./silent"
elf = context.binary

io = process("./silent")

def add(size, cont):
    io.sendline("1")
    io.sendline(str(size))
    io.send(cont)
    sleep(0.01)

def delete(idx):
    io.sendline("2")
    io.sendline(str(idx))

io.interactive()
```
