Test.java

```java
public class Test {
		
	public static boolean is2(int n) {
		if (n <= 3) {
			if (n > 1) {
				return true;
			}
			return false;
		} else if (n % 2 == 0 || n % 3 == 0) {
			return false;
		} else {
			int i = 5;
			while (i * i <= n) {
				if (n % i == 0 || n % (i + 2) == 0) {
					return false;
				}
				i += 6;
			}
			return true;
		}
	}
	
	public static void main(String args[]) {
		int count = Integer.parseInt(args[0]);
		int k = 0;
		while (count > 0) {
			if (is2(count)) {
				k += 100;
			} else {
				k--;
			}
			count--;
		}
		System.out.println(k);
	}
}
```
